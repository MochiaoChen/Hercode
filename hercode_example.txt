# Hello! Her World         # E 魔法首行：启动 HerCode 的温柔模式

# -------------------------------------------------------------
# 「你可以做到 (you_can_do_this)
#  这是一段“鼓励式函数”：
#   - 功能：向终端打印两句话，欢迎女孩来到专属的编程世界
#   - 关键词说明：
#       function     —— 定义函数（像在写日记那样自然）
#       say          —— 输出一句温暖的话
#       end          —— 结束函数或代码块
# -------------------------------------------------------------


function you_can_do_this:
    say "Hello! Her World !"  
    say "编程很美，也属于你！"  
end

# -------------------------------------------------------------
# E 入口区块 start
# HerCode 约定：程序从 start: 开始执行
# -------------------------------------------------------------

start:
    you_can_do_this                
end
